How to Generate OG Images

Method 1: Using Puppeteer (Automated)

const puppeteer = require('puppeteer');

async function generateOGImage(title, category, excerpt) {
  const browser = await puppeteer.launch();
  const page = await browser.newPage();

  // Set viewport to OG image size
  await page.setViewport({ width: 1200, height: 630 });

  // Load this HTML file
  await page.goto('file:///path/to/og-image-template.html');

  // Update content
  await page.evaluate((title, category, excerpt) => {
    document.querySelector('.og-title').textContent = title;
    document.querySelector('.og-category').textContent = category;
    document.querySelector('.og-excerpt').textContent = excerpt;
  }, title, category, excerpt);

  // Take screenshot
  await page.screenshot({
    path: 'og-image.png',
    type: 'png'
  });

  await browser.close();
}

// Generate image
generateOGImage(
  'Getting Started with Claude Code on AWS Bedrock',
  'Tutorial',
  'Learn how to set up and use Claude Code with AWS Bedrock'
);
